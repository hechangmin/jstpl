 10 ;(function(global, undefined){
 11 
 12     var config = {openTag : '<%', closeTag : '%>', escape : true},
 13         cache = {};
 14 
 15     /**
 16      * @name jstpl
 17      * @constructor
 18      * @class  jstpl 模板引擎
 19      * @param  {json} options 配置开始标签，关闭标签,如果第一个参数不是对象，则自动转调 template
 20      * @example jstpl({openTag : '<%', closeTag : '%>', escape : true});
 21      */
 22     var jstpl = function(options) {
 23         if (undefined !== options) {
 24             if('string' === typeof options){
 25                 return jstpl.template.apply(this, arguments);
 26             }
 27             for (var option in options) {
 28                 config[option] = options[option];
 29             }
 30         }
 31     };
 32 
 33     /**
 34      * @description {Sting} 版本
 35      * @field
 36      */
 37     jstpl.version = '1.0.0';
 38 
 39     if (global.jstpl) {
 40         return;
 41     } else {
 42         global.jstpl = jstpl;
 43     }
 44 
 45     /**
 46      * @description template
 47      * @param {string} 元素ID 或 模板内容
 48      * @param {json} data 模板数据 若空，则返回模板函数
 49      * @return 模板解析结果
 50      */
 51     jstpl.template = function(str, data){
 52         var isDomId = !/[^a-zA-Z10-9_-]/.test(str), element, fn, tpl;
 53 
 54         if(isDomId){                    // 是元素id的格式
 55             if(global.document){        // 是浏览器环境
 56                 if(cache[str]){         // 有缓存
 57                     fn = cache[str];
 58                 }else{
 59                     element = document.getElementById(str);
 60                     tpl = element.value || element.innerHTML;
 61                     fn = cache[str] = arguments.callee(tpl)
 62                 }
 63             }else{
 64                 throw 'param error : [non browser, nonsupport element id]';
 65                 return;
 66             }
 67         }else{
 68             fn = compile(str);
 69         }
 70         return data ? fn( data ) : fn;
 71     };
 72 
 73     //模板编译成函数
 74     var compile = function(str){
 75         //避免with语法
 76         var strFn = "var _$jstpl='',__fn__=(function(__d__){var __v__='';for(var __k__ in __d__){__v__+=('var '+__k__+'=__d__[\"'+__k__+'\"];');};eval(__v__);_$jstpl+='" + parse(str) + "';__v__=null;})(param);__fn__ = null;return _$jstpl;";
 77         return new Function("param", strFn);
 78     };
 79 
 80     //转义影响正则的字符
 81     var encodeReg = function (source) {
 82         return String(source).replace(/([.*+?^=!:${}()|[\]/\\])/g,'\\$1');
 83     };
 84 
 85     /**
 86      * @description avoidXSS
 87      * @param {string} 待处理字符串
 88      * @return 转义后的字符串
 89      */
 90     jstpl.avoidXSS = function(strParam) {
 91         var m = {
 92             "<": "<",
 93             ">": ">",
 94             '"': """,
 95             "'": "'",
 96             "&": "&"
 97         };
 98         return String(strParam).replace(/[&<>"']/g, function (s) {
 99             return m[s];
100         });
101     };
102 
103     //模板解析
104     var parse = function(str){
105         var openTag = encodeReg(config['openTag']),
106             closeTag = encodeReg(config['closeTag']),
107             escape = config['escape'];
108 
109         //移除注释及换行，避免干扰正常解析
110         str = String(str).replace(new RegExp("(" + openTag + "[^" + closeTag + "]*)//.*\n","g"), "$1")
111             .replace(new RegExp("<!--.*?-->", "g"),"")
112             .replace(new RegExp("[\\r\\t\\n]","g"), "");
113 
114         return str.replace(new RegExp( openTag + '(.*?)' + closeTag, 'g'), function($, $1){
115 
116             var s = $1,
117                 bFlag = false;
118 
119             if($1.indexOf('=') == 0){
120 
121                 bFlag = true;
122                 s = s.substr(1);
123 
124                 //是否转义
125                 if(escape){
126                     s = 'jstpl.avoidXSS(' + s + ')';
127                 }
128 
129                 s =  '_$jstpl+=' + s + ";";
130             }
131 
132             return "';" + s + "_$jstpl += '";
133         });
134     };
135 
136     if ('function' === typeof define) {
137         define(function() {
138             return jstpl;
139         });
140     } else if ('undefined' !== typeof exports) {
141         module.exports = jstpl;
142     }
143 })(window);
